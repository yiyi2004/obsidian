func BubbleSort[T any](arr []T, comparator func(i T, j T) bool) {
	for i := 0; i < len(arr); i++ {
		// 这里需要注意越界问题
		for j := 0; j < len(arr)-1-i; j++ {
			if comparator(arr[j], arr[j+1]) {
				arr[j], arr[j+1] = arr[j+1], arr[j]
			}
		}
	}
}
