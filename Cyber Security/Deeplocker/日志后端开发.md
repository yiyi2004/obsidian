go func() {  
    // ws://IP:PORT/irm/info  
    http.HandleFunc("/irm/info", func(w http.ResponseWriter, r *http.Request) {  
       conn, err := upgrader.Upgrade(w, r, nil)  
       if err != nil {  
          log.Println(err)  
          return  
       }  
       defer conn.Close()  
  
       for {  
          select {  
          case info := <-ctl.wsc.irmInfoChan.c:  
             err := conn.WriteMessage(websocket.TextMessage, []byte(info))  
             if err != nil {  
                log.Println(err)  
                return  
             }  
          }  
       }  
    })  
  
    http.HandleFunc("/irm/image", func(w http.ResponseWriter, r *http.Request) {
