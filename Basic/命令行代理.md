```shell
$Env:http_proxy="http://127.0.0.1:7890";
```

```shell
export http_proxy='http://localhost:7890'
export https_proxy='http://localhost:7890'
```

```shell
export hostip=$(cat /etc/resolv.conf |grep -oP '(?<=nameserver\ ).*')
export ALL_PROXY="http://${hostip}:7890";
```
