func MergeSort(arr, help []int, l, r int) {
	if l >= r {
		return
	}

	m := (l + r) / 2
	MergeSort(arr, help, l, m)
	MergeSort(arr, help, m+1, r)
	Merge(arr, help, l, m, r)
}

func Merge(arr, help []int, l, m, r int) {
	p1, p2 := l, m+1
	i := l
	for p1 <= m && p2 <= r {
		if arr[p1] < arr[p2] {
			help[i] = arr[p1]
			p1++
		} else {
			help[i] = arr[p2]
			p2++
		}
		i++
	}

	for p1 <= m {
		help[i] = arr[p1]
		p1++
		i++
	}
	for p2 <= r {
		help[i] = arr[p2]
		p2++
		i++
	}
	for i := l; i <= r; i++ {
		arr[i] = help[i]
	}
}
